        function switchSet(num) {

        async function toggleCam(vidId, imgId, btnId) {

            const video = document.getElementById(vidId);

            const img = document.getElementById(imgId);

            const btn = document.getElementById(btnId);


            if (streams[vidId]) {

                streams[vidId].getTracks().forEach(t => t.stop());

                delete streams[vidId];

                video.srcObject = null;

                video.classList.add('hidden');

                img.classList.remove('hidden');

                btn.classList.remove('btn-active');

            } else {

                try {

                    const stream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: 'environment' } });

                    streams[vidId] = stream;

                    video.srcObject = stream;

                    video.classList.remove('hidden');

                    img.classList.add('hidden');

                    btn.classList.add('btn-active');

                } catch (err) { alert("Hardware Error: Unable to initialize optical sensor."); }

            }

        }


        function previewFile(input, imgId, vidId, btnId) {

            const file = input.files[0];

            const reader = new FileReader();

            reader.onload = (e) => {

                const img = document.getElementById(imgId);

                img.src = e.target.result;

                img.classList.remove('hidden');

                img.style.opacity = "1";

                document.getElementById(vidId).classList.add('hidden');

                if(streams[vidId]) { streams[vidId].getTracks().forEach(t => t.stop()); delete streams[vidId]; }

                document.getElementById(btnId).classList.remove('btn-active');

                

                const idNum = parseInt(imgId.replace('img', ''));

                if (idNum % 2 === 0) {

                    const setNum = Math.ceil(idNum / 2);

                    const ghost = document.getElementById(`ghost-${setNum}`);

                    if(ghost) {

                        ghost.src = e.target.result;

                        ghost.classList.remove('hidden');

                    }

                }

            };

            if(file) reader.readAsDataURL(file);

        }


        function updateGhost(setNum, val) {

            const ghost = document.getElementById(`ghost-${setNum}`);

            if(ghost) ghost.style.opacity = val / 100;

        }


        async function runDualAnalysis() {

            const masterDiff = document.getElementById('master-diff');

            const diffText = document.getElementById('diff-text');

            

            const leftId = (activeSet - 1) * 2 + 1;

            const rightId = (activeSet - 1) * 2 + 2;

            const leftImg = document.getElementById(`img${leftId}`);

            const rightImg = document.getElementById(`img${rightId}`);


            if (!leftImg.src.startsWith('data') || !rightImg.src.startsWith('data')) {

                alert("PROTOCOL ERROR: Two valid specimens are required for differential analysis.");

                return;

            }


            masterDiff.classList.remove('hidden');

            diffText.innerHTML = '<div class="flex items-center gap-2"><div class="loading-spinner"></div> CONDUCTING MORPHOLOGICAL COMPARISON...</div>';


            try {

                const prompt = `Perform a high-precision morphological comparison between Specimen A (Left) and Specimen B (Right). 

                Categorize your findings into:

                1. Structural Variance (Stem, Leaf, Flower, or Cap architecture).

                2. Taxonomic Correlation (Likelihood of being look-alikes or the same species).

                3. Diagnostic Markers (Identify specific traits like hairs, gills, or bracts).

                Use scientific terminology where applicable.`;


                const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify({

                        contents: [{

                            parts: [

                                { text: prompt },

                                { inlineData: { mimeType: "image/png", data: leftImg.src.split(',')[1] } },

                                { inlineData: { mimeType: "image/png", data: rightImg.src.split(',')[1] } }

                            ]

                        }]

                    })

                });


                const data = await response.json();

                diffText.innerText = data.candidates?.[0]?.content?.parts?.[0]?.text || "ANALYSIS ABORTED: No data returned.";

            } catch (err) {

                diffText.innerText = "SYSTEM ERROR: API connection failed.";

            }

        }


        function resetAll() {

            location.reload();

        }
